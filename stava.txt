from selenium.webdriver.common.keys import Keys

driver = webdriver.Chrome()
driver.get("https://www.strava.com")
# Find the login button and click it
login_button = driver.find_element_by_xpath("//a[@href='/login']")
login_button.click()

# Find the email and password fields and enter your credentials
email_field = driver.find_element_by_xpath("//input[@name='email']")
email_field.send_keys("<your_email>")
password_field = driver.find_element_by_xpath("//input[@name='password']")
password_field.send_keys("<your_password>")
password_field.send_keys(Keys.RETURN)
